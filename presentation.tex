\section{Overview}
\subsection{Preview}
\begin{frame}
\frametitle{Inefficient Concessions and Mediation}

In conflict scenarios, concessions are sometimes inefficient 
\pause
\begin{itemize}[<+->]
  \item We show inefficient concessions may be preferable if efficient concessions have potential future cost
	\item Cost: concessions used against you by negotiating partner who violates agreement
	\item Can you trust that partner will make peace?
\end{itemize}

\pause
\vskip.2in
New explanation for mediation
\pause
\begin{itemize}[<+->]
  \item Can remove uncertainty about ability of negotiating partner to commit to peace
	\item Removes need for inefficient concessions
\end{itemize}
\end{frame}


\begin{frame}{What we do}

\pause
Start with simple, two-player repeated Prisoners' Dilemma
\pause
\begin{itemize}[<+->]
	\item Asymmetric information about $\de_i$: can partner commit to peace?
\end{itemize}

\pause
\vskip.2in
Add time zero: let partners give concessions with both signaling and material value
\pause
\begin{itemize}[<+->]
	\item Let material value provide help/harm to the giver
	\item Let partners destroy some/all of the material value
\end{itemize}

\pause
\vskip.2in
Mediator removes uncertainty about partner's $\de$

\end{frame}



\begin{frame}{Literature}
\pause
Signaling: Spence (1973), but signal has material value
\pause
		\begin{itemize}
			\item cost/benefit tradeoff differs b/c of $\de_i$, $C_i(g)$
		\end{itemize} 

\pause
Gift-Giving: Camerer (1988), Prendergast $\&$ Stole (2001)
\pause
		\begin{itemize}
			\item Source of inefficiency differs: gifts can be used against giver
		\end{itemize} 

\pause
Conflict: Slantchev (2011), Arena (2013)
\pause
		\begin{itemize}[<+->]
			\item Concessions are costly signals instead of bargaining chip
			\item Commitment to peace, not resolve to fight
			\item Costly signals are concessions instead of proof of resolve
		\end{itemize} 

\pause
Mediation: Fey and Ramsay (2008, 2011), Horner et al. (2010)
\pause
		\begin{itemize}
			\item Information is about ability to commit, not resolve
		\end{itemize} 
\end{frame}



\section{Model}
\subsection{}
\begin{frame}{Timeline}
\pause
\begin{enumerate}[<+->]
	\item[-1.] Nature independently determines types of Country 1 and Country 2: $\de_i \in \left\{\de_h,\de_l\right\}$
	\item[0.] Countries simultaneously give costly concessions: $g_i \in \mathbb{R}_+$
	\item[1--$\infty$.] Countries engage in a simultaneous Prisoners' Dilemma interaction 
\end{enumerate}

\end{frame}


\begin{frame}{Stage Game}
\pause

\begin{center}
\begin{tabular}{|l|c|c|}
  \hline      & Trust & Fight \\ \hline
	 Trust& T, T& -D, T+W \\ \hline
	Fight & T+W,-D& W-D, W-D \\ \hline
\end{tabular}
\end{center}

\vskip.05in
where 
\begin{itemize}
	\item $T\geq0$: Benefit from the other country playing Trust
	\item $W\geq0$: Additional benefit from playing Fight
	\item $D\geq0$: Damages due to the other country playing Fight
\end{itemize}
\vskip.05in
\pause
Assume $T>W-D$
\end{frame}

\begin{frame}
\begin{itemize}[<+->]
	\item Payoffs: sum the discounted stage game payoffs plus any concessions given or received
		\begin{itemize}
			\item e.g. player's $i$'s payoff if both parties give no concession and play ``Fight'' in every period: $$\sum_{t=1}^\infty \de_i^{t-1} (W-D) = \frac{W-D}{1-\de_i}$$
		\end{itemize}
	\item Parameters are common knowledge with the exception of $\delta_i$, which is country i's private information
	\item Social welfare measured as sum of high types' expected utilities
\end{itemize}
\end{frame}


\section{Benchmark}
\subsection{No Money Burning}


\begin{frame}{Benchmark Model}
\pause
Assume two types: $\de_h$ and $\de_l$
\begin{itemize}[<+->]
	\item $\de_h > \de^* > \de_l$ where $\de^*=\frac{W}{T+D}$ is the cutoff for sustaining (Trust,Trust) eqm
	\item $p$: probability of high type
	\item Cost of giving concessions $g$: $g = c_l(g)\geq c_h(g) = g$
\end{itemize}

\pause
\vskip.2in
Some equilibria of interest
\pause
\begin{itemize}[<+->]
	\item Separating without concessions
	\item Separating through concessions
	\item Pool on `Fight'
\end{itemize}
\end{frame}

\begin{frame}{Separating through concessions}
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 2}
  In the best concessions separating equilibrium, high types give the smallest concession necessary to separate. Low types do not give a concession.
\end{beamerboxesrounded}
	
\pause	
\[
  U_h=pg - c_h(g) + \textstyle \frac{\de_h}{(1-\de_h)} \left[pT+(1-p)(W-D)\right]
\]
\pause
\vskip-.1in

	\begin{itemize}[<+->]
		\item The smallest concession is $p(T+D)$
		\item If $p$ is low, high types are better off in the `fight' pooling equilibrium
	\end{itemize}
\end{frame}

\section{Inefficient Concessions}
\subsection{Add Money Burning}
\begin{frame}{Timeline}
\pause
\begin{enumerate}[<+->]
	\item[-1.] Nature independently determines types of Country 1 and Country 2: $\de_i \in \left\{\de_h,\de_l\right\}$
	\item[0.] Concessions
		\begin{enumerate}
			\item[0a.] Countries simultaneously give costly concessions and decide on the efficiency of the concessions: $(g_i, e_i) \in \mathbb{R}_+ \times [0,1]$ 
			\item[0b.] \textcolor{red!75!green!50!blue!25!gray}{Countries decide what proportion of a received concession to invest in civil society (vs. military capabilities): $\alpha_i \in [0,1]$}
		\end{enumerate}	
	\item[1--$\infty$.] Countries engage in a simultaneous Prisoners' Dilemma interaction 
\end{enumerate}
\end{frame}

\begin{frame}{Burning money is unattractive}
\pause
Now benefit of concession is $eg$

\vskip.1in
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 3}
  When concessions confer no material help or harm on the giver, it is optimal to give efficient gifts ($e=1$) in a separating eqm.
\end{beamerboxesrounded}
\pause
\begin{itemize}[<+->]
	\item The benefit of the gift appears on both sides of the incentive constraint for both individuals, so cancels out
	\item Costs of giving a concession don't change
	\item The benefit appears in the high type's expected utility
\end{itemize}
\end{frame}

\begin{frame}[label=hurt]
\frametitle{Concessions can hurt the giver}
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 4}
  When the low type country can use concessions to reduce the payoffs of its negotiating partner during a `fight' stage (i.e. $\alpha_L < 1$), there are parameters under which the separating-through-concessions equilibrium features inefficient concessions.
\end{beamerboxesrounded}
\hyperlink{Supplemental}{\beamerbutton{Modified Payoffs}}
\pause
\[
  U_h=peg - c_h(g) + \textstyle \frac{\de_h}{1-\de_h} \left[pT(1+eg)+(1-p)(W-D(1+eg))\right]
\]
\pause
\vskip-.1in
\begin{itemize}[<+->]
	\item If $p$ is low, concessions likely to be used against you, so remove material value
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Peace not possible in some scenarios}
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 5}
  When $\alpha_L < 1$ is allowed, peace becomes unachievable for some parameters under which it is achievable when concessions do not have material value.
\end{beamerboxesrounded}

\pause
\vskip.1in
\begin{itemize}[<+->]
	\item If peace is achievable, concessions may be either efficient or inefficient
	\item High-type utility may increase or decrease from no-material-value case
\end{itemize}
\end{frame}


\section{Mediation}
\subsection{}
\begin{frame}{Mediator as mechanism designer}
\pause
`Manipulative' mediator: parties report their types, must deliver the stipulated concessions
\pause
\begin{itemize}[<+->]
  \item Mechanism: if two high types, give concession and play `Trust'; Otherwise, no concession and `Fight'
	\item Concession is necessary to get truthful revelation, but only to high type
	\item Need cost of concession for low type to be not to large relative to cost for high type
\end{itemize}
\vskip.1in
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 6}
  A mediator restores peace where $\alpha_i < 1$ destroys it, eliminates inefficient concessions, and reduces the threshold $\de_h$ for high types.
\end{beamerboxesrounded}

\end{frame}
