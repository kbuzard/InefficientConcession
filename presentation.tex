\section{Overview}
\subsection{Preview}
\begin{frame}
\frametitle{Inefficient Concessions and Mediation}

In conflict scenarios, concessions are sometimes inefficient 
\pause
\begin{itemize}[<+->]
  \item We show inefficient concessions may be preferable if efficient concessions have potential future cost
	\item Cost: concessions used against you by negotiating partner who violates agreement
	\item Can you trust that partner will make peace?
\end{itemize}

\pause
\vskip.2in
New explanation for mediation
\pause
\begin{itemize}[<+->]
  \item Can remove uncertainty about ability of negotiating partner to commit to peace
	\item Removes need for inefficient concessions
\end{itemize}
\end{frame}


\begin{frame}{What we do}

\pause
Start with simple, two-player repeated Prisoners' Dilemma
\pause
\begin{itemize}[<+->]
	\item Asymmetric information about partner's discount factor $\de_i$: can partner commit to peace?
\end{itemize}

\pause
\vskip.2in
Add time zero: let partners give concessions with both signaling and material value
\pause
\begin{itemize}[<+->]
	\item Let material value provide help/harm to the giver
	\item Let partners destroy some/all of the material value
\end{itemize}

\pause
\vskip.2in
Mediator removes uncertainty about partner's $\de$

\end{frame}



\begin{frame}{Literature}
\pause
Signaling: Spence (1973), but signal has potential future cost
\pause
		\begin{itemize}
			\item cost/benefit tradeoff b/c of $\de_i$, $C_i(g)$
		\end{itemize} 

\pause
Gift-Giving: Camerer (1988), Prendergast $\&$ Stole (2001)
\pause
		\begin{itemize}
			\item Source of inefficiency differs: gifts can be used against giver
		\end{itemize} 

\pause
Conflict: Slantchev (2011), Arena (2013)
\pause
		\begin{itemize}[<+->]
			\item Concessions are costly signals instead of bargaining chip
			\item Commitment to peace, not resolve to fight
			\item Costly signals are concessions instead of proof of resolve
		\end{itemize} 

\pause
Mediation: Fey and Ramsay (2008, 2011), Horner et al. (2010)
\pause
		\begin{itemize}
			\item Information is about ability to commit, not resolve
		\end{itemize} 
\end{frame}



\section{Model}
\subsection{}
\begin{frame}{Timeline}
\pause
\begin{enumerate}[<+->]
	\item[-1.] Nature independently determines types of Country 1 and Country 2: $\de_i \in \left\{\de_h,\de_l\right\}$
	\item[0.] Countries simultaneously give costly concessions: $g_i \in \mathbb{R}_+$
	\item[1--$\infty$.] Countries engage in a simultaneous Prisoners' Dilemma interaction 
\end{enumerate}

\end{frame}


\begin{frame}{Stage Game}
\pause

\begin{center}
\begin{tabular}{|l|c|c|}
  \hline      & Trust & Fight \\ \hline
	 Trust& T, T& -D, T+W \\ \hline
	Fight & T+W,-D& W-D, W-D \\ \hline
\end{tabular}
\end{center}

\vskip.05in
where 
\begin{itemize}
	\item $T\geq0$: Benefit from the other country playing Trust
	\item $W\geq0$: Additional benefit from playing Fight
	\item $D\geq0$: Damages due to the other country playing Fight
\end{itemize}
\vskip.05in
\pause
Assume $T>W-D$
\end{frame}

\begin{frame}
\begin{itemize}[<+->]
	\item Payoffs: sum the discounted stage game payoffs plus any concessions given or received
		\begin{itemize}
			\item e.g. player's $i$'s payoff if both parties give no concession and play ``Fight'' in every period: $$\sum_{t=1}^\infty \de_i^{t-1} (W-D) = \frac{W-D}{1-\de_i}$$
		\end{itemize}
	\item Parameters are common knowledge with the exception of $\delta_i$, which is country i's private information
	\item Social welfare measured as sum of high types' expected utilities
\end{itemize}
\end{frame}


\section{Benchmark}
\subsection{No Money Burning, No Impact on Giver}


\begin{frame}{Benchmark Model}
\pause
Assume two types: $\de_h$ and $\de_l$
\begin{itemize}[<+->]
	\item $\de_h > \de^* > \de_l$ where $\de^*=\frac{W}{T+D}$ is the cutoff for sustaining (Trust,Trust) eqm
	\item $p$: probability of high type
	\item Cost of giving concessions $g$: $g = c_l(g)\geq c_h(g) = g$
\end{itemize}

\pause
\vskip.2in
Some equilibria of interest
\pause
\begin{itemize}[<+->]
	\item Pool on `Fight'
	\item Separating through concessions
	\item Separating without concessions
\end{itemize}
\end{frame}

\begin{frame}{Pooling Equilibrium}
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Lemma 4}
  From period 1 on, playing fight in all periods is the only sequentially rational strategy for low types regardless of their beliefs of the other country's type and strategy.
\end{beamerboxesrounded}
	
\pause
\vskip.1in

	\begin{itemize}[<+->]
		\item This equilibrium can always be chosen by both types
	\end{itemize}
\end{frame}


\begin{frame}{Separating through concessions}
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 2}
  In the best concessions separating equilibrium, high types give the smallest concession necessary to separate. Low types do not give a concession.
\end{beamerboxesrounded}
	
\pause	
\vskip.1in
If $p$ is low enough, high types are better off in the `fight' pooling equilibrium
\pause
\[
  U_h(CSE)=pg - g + \textstyle \frac{1}{(1-\de_h)} \left[pT+(1-p)(W-D)\right]
\]
\pause
\[
  U_h(POOL)=\textstyle \frac{1}{(1-\de_h)} \left(W-D\right)
\]

\end{frame}


\begin{frame}{Smallest Separating Concession}
The smallest concession, $p(T+D)$, comes from the low type's incentive constraint
\pause
\[
  pg +X_{FF}^l \geq -g + pg + pX_{FT}^l+(1-p)X_{FF}^l 
\]
\pause
\[
  pg +\frac{W-D}{1-\de_l} \geq -g + pg + p\left[T+W + \frac{\de \left(W-D\right)}{1-\de_l} \right]+(1-p)\frac{W-D}{1-\de_l}
\]
\pause
\[
  g \geq p\left[T+W + \frac{\de \left(W-D\right)}{1-\de_l} \right]-p\frac{\left(W-D\right)}{1-\de_l}
\]
\pause
\[
  g \geq pT+pW -pW -pD + p\frac{\left(W-D\right)}{1-\de_l} -p\frac{\left(W-D\right)}{1-\de_l}
\]
\end{frame}

\section{Inefficient Concessions}
\subsection{Add Impact of Concessions on Giver (Still no money burning)}
\begin{frame}[label = hurt]{Timeline}
\pause
\begin{enumerate}[<+->]
	\item[-1.] Nature independently determines types of Country 1 and Country 2: $\de_i \in \left\{\de_h,\de_l\right\}$
	\item[0.] Concessions
		\begin{enumerate}
			\item[0a.] Countries decide what proportion of a received concession to invest in civil society (vs. military capabilities): $\alpha_i \in [0,1]$ \hskip.2in \hyperlink{Supplemental}{\beamerbutton{Modified Payoffs}}
			\item[0b.] \textcolor{red!75!green!50!blue!25!gray}{Countries simultaneously give costly concessions and decide on the efficiency of the concessions: $(g_i, e_i) \in \mathbb{R}_+ \times [0,1]$} 
		\end{enumerate}	
	\item[1--$\infty$.] Countries engage in a simultaneous Prisoners' Dilemma interaction 
\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Peace not possible in some scenarios}
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 3}
  When concessions impact the giver's future welfare, peace becomes unachievable for some parameters under which it is achievable when concessions do not impact the giver's future welfare.
\end{beamerboxesrounded}

\pause
\vskip.1in
\begin{itemize}[<+->]
	\item Effect comes in part through change in minimum separating concession
	\item High-type utility may increase or decrease from the benchmark case
\end{itemize}
\end{frame}


\subsection{Add Money Burning}
\begin{frame}{Timeline}
\pause
\begin{enumerate}[<+->]
	\item[-1.] Nature independently determines types of Country 1 and Country 2: $\de_i \in \left\{\de_h,\de_l\right\}$
	\item[0.] Concessions
		\begin{enumerate}
			\item[0a.] Countries decide what proportion of a received concession to invest in civil society (vs. military capabilities): $\alpha_i \in [0,1]$
			\item[0b.] \textcolor{blue}{Countries simultaneously give costly concessions and decide on the efficiency of the concessions: $(g_i, e_i) \in \mathbb{R}_+ \times [0,1]$}
		\end{enumerate}	
	\item[1--$\infty$.] Countries engage in a simultaneous Prisoners' Dilemma interaction 
\end{enumerate}
\end{frame}

\begin{frame}{Burning money is (obviously) unattractive...}
\pause
Now benefit of concession is $eg$ 

\vskip.1in
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Lemma 2}
  If a concessions separating equilibrium exists under no money burning, it is optimal to give efficient gifts ($e=1$) in a concessions separating equilibrium with money burning.
\end{beamerboxesrounded}
\pause
\begin{itemize}[<+->]
	\item The benefit of the gift appears on both sides of the incentive constraint for both individuals, so cancels out
	\item Costs of giving a concession don't change
	\item The benefit appears in the high type's expected utility
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{...but in some cases countries will burn money}
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 4}
  When concessions affect the future welfare of the giver \textit{and} their value can be destroyed, there are parameters under which the optimal equilibrium is a separating equilibrium in which concessions are inefficient.
\end{beamerboxesrounded}
\vskip-.1in
\pause
\[
  U_h=peg - c_h(g) + \textstyle \frac{1}{1-\de_h} \left[pT(1+eg)+(1-p)(W-D(1+eg))\right]
\]
\pause
\vskip-.1in
\begin{itemize}[<+->]
	\item If $p$ is low, concessions likely to be used against you
	\item If $\de_l$ is low, concession has to be large to deter low type from from mimicking high type
\end{itemize}
\end{frame}


\section{Mediation}
\subsection{}
\begin{frame}{Mediator as mechanism designer}
\pause
`Manipulative' mediator: parties report their types, must deliver the stipulated concessions
\pause
\begin{itemize}[<+->]
  \item Mechanism: if two high types, give concession and play `Trust'; Otherwise, no concession and `Fight'
	\item Concession is necessary to get truthful revelation, but only to high type
	\item Need cost of concession for low type to be not too large relative to cost for high type
\end{itemize}
\vskip.1in
\pause
\begin{beamerboxesrounded}[upper=palette tertiary, shadow=true]{Theorem 5}
  A mediator restores peace where concessions' future welfare impact destroys it and eliminates inefficient concessions elsewhere.
\end{beamerboxesrounded}

\end{frame}
